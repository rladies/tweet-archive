@minebocek For examples, here are 5 ways to get the mean of each row of a df, as a vector

df <- data.frame(x = 5:7, y = 1:3)

1) *apply:
apply(df, 1, FUN = mean)

2) for loop:
mean <- rep(NA, nrow(df))

for (r in 1:nrow(df)) {
mean[r] <- mean(df[r, , drop = TRUE] %>% unlist())
}
